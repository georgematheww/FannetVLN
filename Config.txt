<?php
// config.php - DB connection (FANNET lab)
$DB_HOST = '127.0.0.1';
$DB_USER = 'root';
$DB_PASS = ''; // default XAMPP root password; if you set one, update here
$DB_NAME = 'fannet_lab';

ini_set('display_errors', 1);
error_reporting(E_ALL);

$conn = new mysqli($DB_HOST, $DB_USER, $DB_PASS);
if ($conn->connect_error) die("Connect err: " . $conn->connect_error);
$conn->query("CREATE DATABASE IF NOT EXISTS $DB_NAME");
$conn->select_db($DB_NAME);
